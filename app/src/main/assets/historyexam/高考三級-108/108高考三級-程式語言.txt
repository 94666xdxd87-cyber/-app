class Shape: 
def __init__(self, x, y): 
    self.x = x 
    self.y = y 
    self.description = "unknown" 
def area(self): 
    return self.x * self.y 
def perimeter(self): 
    return 2 * self.x + 2 * self.y 
def describe(self, text): 
    self.description = text
